135 inline void interpolate_orientation( float time_lag, const DynamicObject* object, DirectX::XMVECTOR& result_vector )
136 {
137  XMVECTOR orientation = object->get_orientation( );
138  XMVECTOR rotation_velocity = object->get_rotation_speed( );
139 
140 #ifdef _QUATERNION_SPEED
141  //najpierw liczymy nowy kwaternion dla obrotu w czasie sekundy
142  rotation_velocity = XMQuaternionMultiply( orientation, rotation_velocity );
143  //teraz interpolujemy poprzednią orientację i orientację po sekundzie
144  //ze wspóczynnikiem równym czasowi jaki upłynął faktycznie
145  result_vector = XMQuaternionSlerp( orientation, rotation_velocity, time_lag );
146 
147  /*Dużo obliczeń, może da się to jakoś załatwić bez interpolacji...*/
148 #else
149 
150 
151  if ( !XMVector3Equal( rotation_velocity, XMVectorZero() ) )
152  {
153  float rot_angle = XMVectorGetW( rotation_velocity ) * time_lag; // liczymy kąt obrotu
154  rotation_velocity = XMQuaternionRotationAxis( rotation_velocity, rot_angle ); // przerabiamy na kwaternion
155  result_vector = XMQuaternionMultiply( orientation, rotation_velocity ); // liczymy nowy kwaternion orientacji
156  }
157  else
158  result_vector = orientation;
159 #endif
160 }
