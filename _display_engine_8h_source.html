164 inline void interpolate_orientation( float time_lag, const DynamicActor* object, DirectX::XMVECTOR& result_vector )
165 {
166  DirectX::XMVECTOR orientation = object->GetOrientation( );
167  DirectX::XMVECTOR rotation_velocity = object->GetRotationSpeed( );
168 
169 #ifdef _QUATERNION_SPEED
170  //najpierw liczymy nowy kwaternion dla obrotu w czasie sekundy
171  rotation_velocity = XMQuaternionMultiply( orientation, rotation_velocity );
172  //teraz interpolujemy poprzednią orientację i orientację po sekundzie
173  //ze wspóczynnikiem równym czasowi jaki upłynął faktycznie
174  result_vector = XMQuaternionSlerp( orientation, rotation_velocity, time_lag );
175 
176  /*Dużo obliczeń, może da się to jakoś załatwić bez interpolacji...*/
177 #else
178 
179 
180  if ( !DirectX::XMVector3Equal( rotation_velocity, DirectX::XMVectorZero() ) )
181  {
182  float rot_angle = DirectX::XMVectorGetW( rotation_velocity ) * time_lag; // liczymy kąt obrotu
183  rotation_velocity = DirectX::XMQuaternionRotationAxis( rotation_velocity, rot_angle ); // przerabiamy na kwaternion
184  result_vector = DirectX::XMQuaternionMultiply( orientation, rotation_velocity ); // liczymy nowy kwaternion orientacji
185  }
186  else
187  result_vector = orientation;
188 #endif
189 }
